  \begin{displaymath}
    Σ = \conv\{v_0,\dots,v_n\}, 
    \end{displaymath}
    where $v_0,\dots,v_n ∈ ℝ^n$ are affinely independent, i.e. $v_1-v_0, \dots,v_n - v_0$ are linearly independent.

  \begin{enumerate}
  \item   there exist $n+1$ vertices of $P$ that are affinely independent. 
  \end{enumerate}
