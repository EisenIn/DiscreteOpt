 \begin{equation*}

\begin{frame}[t]
  \frametitle{$∃$ over finite set: More details }

  $\cone(A) = \{ Ax ： x∈ ℝ_{≥0}^n\} $; \tabb   $\intcone(A) = \{ Ax ： x∈ ℤ_{≥0}^n\}  $


  \bigskip 
   \begin{exampleblock}{$∀-∃$-statement}%{Final step }
  Given: \tabb  $Q ⊆ ℝ^m$ $A ∈ ℤ^{m ×n}$ 

  Decide: 
 \begin{equation*}
   ∀ b ∈Q ∩ ℤ^m \quad ： \quad b ∈ \intcone(A)
 \end{equation*}
\end{exampleblock}
  
  \uncover<2->{
   \bigskip
 
   If $A ∈ ℤ^{m ×m}$ nonsingular: $\intcone(A) = \cone(A) ∩ Λ(A)$ -- \mred{ integer programming}
 }
\end{frame}

\begin{frame}[t] 
  \frametitle{Proximity}
  \begin{theorem}[E. \& Weismantel (2019)]
  \label{thm:prox}
Given: 
$
\label{eq:ProximityIP}
  \max \big\{ c^T x : Ax = b, \, x ≥ {0}, \, x ∈ ℤ^n \big\}  
$

\medskip 
\tabb $x^*∈ℝ^n_{\geq 0}$:  optimal fractional vertex

\medskip 
There exists optimal $z^*∈ℤ^n_{\geq 0}$ with
\begin{displaymath}
  \|z^* - x^* \|_1 \leq m(2m \Delta + 1)^m 
\end{displaymath}
\end{theorem}
\end{frame}
